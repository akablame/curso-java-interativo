Classe Retângulo
Iniciante

Crie uma classe Retangulocom atributos largurae altura. Adicione métodos para calcular a áreae o perímetro, e um método exibir()que mostra tudo formatado.

Dica

Área = largura × altura. Perímetro = 2 × (largura + altura).

Java
public class Retangulo {
 private double largura;
 private double altura;

 public Retangulo(double largura, double altura) {
 this.largura = largura;
 this.altura = altura;
 }

 public double calcularArea() {
 return largura * altura;
 }

 public double calcularPerimetro() {
 return 2 * (largura + altura);
 }

 public void exibir() {
 System.out.println("=== Retângulo ===");
 System.out.printf("Dimensões: %.1f x %.1f%n", largura, altura);
 System.out.printf("Área: %.2f%n", calcularArea());
 System.out.printf("Perímetro: %.2f%n", calcularPerimetro());
 }
}

// Main
Retangulo r = new Retangulo(5.0, 3.0);
r.exibir();
// === Retângulo ===
// Dimensões: 5.0 x 3.0
// Área: 15.00
// Perímetro: 16.00

Classe Aluno com Encapsulamento
Encapsulamento

Crie uma classe Alunocom atributos privados: nome, matrícula e array de notas (5 notas). Use getters/setters. Adicione métodos: calcularMedia()e getSituacao()(Aprovado/Recuperação/Reprovado).

Dica

Use um loop for-each para somar as notas. Divida pela quantidade. No setter de nota, valide que o valor é entre 0 e 10.

Java
public class Aluno {
 private String nome;
 private String matricula;
 private double[] notas;

 public Aluno(String nome, String matricula) {
 this.nome = nome;
 this.matricula = matricula;
 this.notas = new double[5];
 }

 public String getNome() { return nome; }
 public String getMatricula() { return matricula; }

 public void setNota(int posicao, double valor) {
 if (posicao >= 0 && posicao < 5 && valor >= 0 && valor <= 10) {
 notas[posicao] = valor;
 } else {
 System.out.println(" Nota ou posição inválida!");
 }
 }

 public double calcularMedia() {
 double soma = 0;
 for (double n : notas) soma += n;
 return soma / notas.length;
 }

 public String getSituacao() {
 double media = calcularMedia();
 if (media >= 7.0) return " Aprovado";
 if (media >= 5.0) return " Recuperação";
 return " Reprovado";
 }

 public void boletim() {
 System.out.println("=== BOLETIM: " + nome + " (" + matricula + ") ===");
 for (int i = 0; i < notas.length; i++) {
 System.out.printf("Nota %d: %.1f%n", i+1, notas[i]);
 }
 System.out.printf("Média: %.2f — %s%n", calcularMedia(), getSituacao());
 }
}

// Main
Aluno a = new Aluno("Ana Silva", "2024001");
a.setNota(0, 8.5);
a.setNota(1, 7.0);
a.setNota(2, 9.0);
a.setNota(3, 6.5);
a.setNota(4, 8.0);
a.boletim();

Hierarquia de Veículos
Herança

Crie a hierarquia: Veiculo(marca, ano, velocidadeMaxima) → Carro(qtdPortas) e Moto(cilindrada). Cada subclasse deve sobrescrever o método exibirInfo().

Dica

Na subclasse, use super.exibirInfo()para chamar o método do pai primeiro, depois adicione as informações específicas.

Java
public class Veiculo {
 protected String marca;
 protected int ano;
 protected int velocidadeMaxima;

 public Veiculo(String marca, int ano, int velMax) {
 this.marca = marca;
 this.ano = ano;
 this.velocidadeMaxima = velMax;
 }

 public void exibirInfo() {
 System.out.println("Marca: " + marca);
 System.out.println("Ano: " + ano);
 System.out.println("Vel. Máx: " + velocidadeMaxima + " km/h");
 }
}

public class Carro extends Veiculo {
 private int qtdPortas;

 public Carro(String marca, int ano, int velMax, int portas) {
 super(marca, ano, velMax);
 this.qtdPortas = portas;
 }

 @Override
 public void exibirInfo() {
 System.out.println(" === CARRO ===");
 super.exibirInfo();
 System.out.println("Portas: " + qtdPortas);
 }
}

public class Moto extends Veiculo {
 private int cilindrada;

 public Moto(String marca, int ano, int velMax, int cilindrada) {
 super(marca, ano, velMax);
 this.cilindrada = cilindrada;
 }

 @Override
 public void exibirInfo() {
 System.out.println(" === MOTO ===");
 super.exibirInfo();
 System.out.println("Cilindrada: " + cilindrada + "cc");
 }
}

// Polimorfismo!
Veiculo[] frota = {
 new Carro("Honda", 2022, 180, 4),
 new Moto("Yamaha", 2023, 200, 600),
 new Carro("Ford", 2021, 160, 2)
};
for (Veiculo v : frota) { v.exibirInfo(); System.out.println(); }

Interface Calculável
Interface

Crie uma interface Calculavelcom métodos calcularArea()e calcularPerimetro(). Implemente-a nas classes Circuloe Triangulo.

Dica

Círculo: área = π × r². Triângulo (equilátero): área = (√3/4) × lado². Use Math.PIe Math.sqrt(3).

Java
public interface Calculavel {
 double calcularArea();
 double calcularPerimetro();
}

public class Circulo implements Calculavel {
 private double raio;
 public Circulo(double raio) { this.raio = raio; }

 @Override
 public double calcularArea() {
 return Math.PI * raio * raio;
 }

 @Override
 public double calcularPerimetro() {
 return 2 * Math.PI * raio;
 }

 public void exibir() {
 System.out.printf(" Círculo (r=%.1f): Área=%.2f | Perímetro=%.2f%n",
 raio, calcularArea(), calcularPerimetro());
 }
}

public class Triangulo implements Calculavel {
 private double lado; // equilátero

 public Triangulo(double lado) { this.lado = lado; }

 @Override
 public double calcularArea() {
 return (Math.sqrt(3) / 4) * lado * lado;
 }

 @Override
 public double calcularPerimetro() {
 return 3 * lado;
 }

 public void exibir() {
 System.out.printf(" Triângulo (l=%.1f): Área=%.2f | Perímetro=%.2f%n",
 lado, calcularArea(), calcularPerimetro());
 }
}

// Polimorfismo com interface
Calculavel[] formas = { new Circulo(5), new Triangulo(6) };
for (Calculavel f : formas) {
 System.out.printf("Área=%.2f | Perímetro=%.2f%n", f.calcularArea(), f.calcularPerimetro());
}

Sistema de Biblioteca (Mini-Projeto)
Completo

Crie as classes: Livro(titulo, autor, isbn, disponivel) e Biblioteca(com array de Livros). A biblioteca deve ter métodos: adicionarLivro(), emprestarLivro(isbn), devolverLivro(isbn)e listarLivros().

Dica

Use um ArrayList ou array de Livro na Biblioteca. Para buscar por ISBN, percorra o array comparando com .equals().

Java
public class Livro {
 private String titulo;
 private String autor;
 private String isbn;
 private boolean disponivel;

 public Livro(String titulo, String autor, String isbn) {
 this.titulo = titulo;
 this.autor = autor;
 this.isbn = isbn;
 this.disponivel = true;
 }

 public String getIsbn() { return isbn; }
 public boolean isDisponivel() { return disponivel; }
 public void setDisponivel(boolean d) { this.disponivel = d; }

 public void exibir() {
 String status = disponivel ? " Disponível" : " Emprestado";
 System.out.printf(" %-25s | %-15s | ISBN: %-12s | %s%n",
 titulo, autor, isbn, status);
 }
}

public class Biblioteca {
 private Livro[] acervo;
 private int quantidade;
 private static final int CAPACIDADE = 100;

 public Biblioteca() {
 this.acervo = new Livro[CAPACIDADE];
 this.quantidade = 0;
 }

 public void adicionarLivro(Livro livro) {
 if (quantidade < CAPACIDADE) {
 acervo[quantidade++] = livro;
 System.out.println(" Livro adicionado: " + livro.getIsbn());
 } else {
 System.out.println(" Biblioteca cheia!");
 }
 }

 public boolean emprestarLivro(String isbn) {
 for (int i = 0; i < quantidade; i++) {
 if (acervo[i].getIsbn().equals(isbn)) {
 if (acervo[i].isDisponivel()) {
 acervo[i].setDisponivel(false);
 System.out.println(" Livro emprestado com sucesso!");
 return true;
 } else {
 System.out.println(" Livro já está emprestado!");
 return false;
 }
 }
 }
 System.out.println(" Livro não encontrado!");
 return false;
 }

 public void devolverLivro(String isbn) {
 for (int i = 0; i < quantidade; i++) {
 if (acervo[i].getIsbn().equals(isbn)) {
 acervo[i].setDisponivel(true);
 System.out.println(" Livro devolvido com sucesso!");
 return;
 }
 }
 System.out.println(" ISBN não encontrado!");
 }

 public void listarLivros() {
 System.out.println("=== ACERVO DA BIBLIOTECA ===");
 for (int i = 0; i < quantidade; i++) {
 acervo[i].exibir();
 }
 }
}

// Main
Biblioteca bib = new Biblioteca();
bib.adicionarLivro(new Livro("Clean Code", "Robert C. Martin", "978-0132350884"));
bib.adicionarLivro(new Livro("Java Efetivo", "Joshua Bloch", "978-0134685991"));
bib.adicionarLivro(new Livro("Design Patterns", "Gang of Four", "978-0201633610"));

bib.listarLivros();
bib.emprestarLivro("978-0132350884");
bib.listarLivros();
