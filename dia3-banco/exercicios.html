1
Criar Tabela de Produtos
CREATE TABLE

Crie uma tabela produtoscom: id (PK auto), nome (VARCHAR 100, obrigatório), preco (DECIMAL 8,2), categoria (VARCHAR 50), estoque (INT, padrão 0), ativo (BOOLEAN, padrão true).

SQL
CREATE TABLE produtos (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(100) NOT NULL,
 preco DECIMAL(8,2),
 categoria VARCHAR(50),
 estoque INT DEFAULT 0,
 ativo BOOLEAN DEFAULT TRUE
);

-- Verificar estrutura criada:
DESCRIBE produtos;
2
Inserir e Consultar Produtos
INSERT + SELECT

Insira 5 produtos na tabela criada. Depois escreva SELECTs para: 1) listar todos, 2) filtrar por categoria "Eletrônicos", 3) listar os 3 mais caros, 4) calcular média de preços.

SQL
-- Inserir produtos
INSERT INTO produtos (nome, preco, categoria, estoque) VALUES
 ('Notebook Dell', 3500.00, 'Eletrônicos', 10),
 ('Mouse Logitech', 89.90, 'Eletrônicos', 50),
 ('Cadeira Gamer', 1200.00, 'Móveis', 5),
 ('Teclado Mecânico', 350.00, 'Eletrônicos', 30),
 ('Mesa de Escritório', 800.00, 'Móveis', 8);

-- 1. Listar todos
SELECT * FROM produtos;

-- 2. Filtrar por categoria
SELECT nome, preco FROM produtos
WHERE categoria = 'Eletrônicos'
ORDER BY preco DESC;

-- 3. Top 3 mais caros
SELECT nome, preco FROM produtos
ORDER BY preco DESC LIMIT 3;

-- 4. Estatísticas
SELECT
 COUNT(*) AS total_produtos,
 AVG(preco) AS media_preco,
 MAX(preco) AS mais_caro,
 MIN(preco) AS mais_barato
FROM produtos;
3
UPDATE e DELETE com Cuidado
UPDATE + DELETE

1. Atualize o preço do Notebook para R$ 3200. 2. Dê 10% de desconto em todos os Eletrônicos. 3. Desative (ativo = false) produtos com estoque = 0. 4. Delete o produto com id = 5.

SQL
-- 1. Atualizar preço específico
UPDATE produtos SET preco = 3200.00 WHERE nome = 'Notebook Dell';

-- 2. Desconto de 10% em Eletrônicos
UPDATE produtos SET preco = preco * 0.90
WHERE categoria = 'Eletrônicos';

-- 3. Desativar sem estoque
UPDATE produtos SET ativo = FALSE WHERE estoque = 0;

-- 4. Deletar por ID (sempre prefira por ID!)
-- Primeiro verificar o que vai ser deletado:
SELECT * FROM produtos WHERE id = 5;
-- Depois deletar:
DELETE FROM produtos WHERE id = 5;
4
JOIN — Pedidos e Clientes
JOIN

Crie tabelas clientese pedidos. Insira dados de exemplo e escreva um JOIN que mostre: nome do cliente, produto do pedido e valor total.

SQL
-- Criar tabelas
CREATE TABLE clientes (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(100) NOT NULL,
 email VARCHAR(150) UNIQUE
);

CREATE TABLE pedidos (
 id INT PRIMARY KEY AUTO_INCREMENT,
 cliente_id INT NOT NULL,
 produto VARCHAR(100),
 valor DECIMAL(8,2),
 data_pedido DATE DEFAULT (CURRENT_DATE),
 FOREIGN KEY (cliente_id) REFERENCES clientes(id)
);

-- Inserir dados
INSERT INTO clientes (nome, email) VALUES
 ('Ana Silva', 'ana@email.com'),
 ('Carlos Lima', 'carlos@email.com'),
 ('Maria Costa', 'maria@email.com');

INSERT INTO pedidos (cliente_id, produto, valor) VALUES
 (1, 'Notebook', 3500.00),
 (1, 'Mouse', 89.90),
 (2, 'Teclado', 350.00),
 (3, 'Monitor', 1200.00);

-- JOIN: clientes com seus pedidos
SELECT
 c.nome AS cliente,
 p.produto,
 p.valor,
 p.data_pedido
FROM pedidos p
INNER JOIN clientes c ON p.cliente_id = c.id
ORDER BY c.nome;

-- Total gasto por cliente
SELECT c.nome, COUNT(p.id) AS total_pedidos, SUM(p.valor) AS total_gasto
FROM clientes c
LEFT JOIN pedidos p ON c.id = p.cliente_id
GROUP BY c.id, c.nome
ORDER BY total_gasto DESC;

Java + MySQL: DAO de Produtos
Java JDBC

Crie uma classe ProdutoDAOem Java com os métodos: inserir(), buscarTodos(), buscarPorId(), atualizar()e deletar()usando JDBC.

Java
import java.sql.*;
import java.util.ArrayList;
import java.util.List;

public class ProdutoDAO {

 public void inserir(String nome, double preco, String categoria) {
 String sql = "INSERT INTO produtos (nome, preco, categoria) VALUES (?, ?, ?)";
 try (Connection c = Conexao.conectar();
 PreparedStatement ps = c.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS)) {
 ps.setString(1, nome);
 ps.setDouble(2, preco);
 ps.setString(3, categoria);
 ps.executeUpdate();
 ResultSet keys = ps.getGeneratedKeys();
 if (keys.next()) System.out.println(" Inserido com id=" + keys.getInt(1));
 } catch (SQLException e) { e.printStackTrace(); }
 }

 public ListbuscarTodos() {
 Listlista = new ArrayList<>();
 String sql = "SELECT * FROM produtos ORDER BY nome";
 try (Connection c = Conexao.conectar();
 PreparedStatement ps = c.prepareStatement(sql);
 ResultSet rs = ps.executeQuery()) {
 while (rs.next()) {
 lista.add(String.format("ID:%d | %s | R$%.2f | %s",
 rs.getInt("id"), rs.getString("nome"),
 rs.getDouble("preco"), rs.getString("categoria")));
 }
 } catch (SQLException e) { e.printStackTrace(); }
 return lista;
 }

 public void atualizar(int id, double novoPreco) {
 String sql = "UPDATE produtos SET preco = ? WHERE id = ?";
 try (Connection c = Conexao.conectar();
 PreparedStatement ps = c.prepareStatement(sql)) {
 ps.setDouble(1, novoPreco);
 ps.setInt(2, id);
 int rows = ps.executeUpdate();
 System.out.println(" " + rows + " produto(s) atualizado(s)");
 } catch (SQLException e) { e.printStackTrace(); }
 }

 public void deletar(int id) {
 String sql = "DELETE FROM produtos WHERE id = ?";
 try (Connection c = Conexao.conectar();
 PreparedStatement ps = c.prepareStatement(sql)) {
 ps.setInt(1, id);
 ps.executeUpdate();
 System.out.println(" Produto id=" + id + " removido");
 } catch (SQLException e) { e.printStackTrace(); }
 }

 // Main para testar
 public static void main(String[] args) {
 ProdutoDAO dao = new ProdutoDAO();
 dao.inserir("Notebook", 3500.00, "Eletrônicos");
 dao.inserir("Mouse", 89.90, "Eletrônicos");
 dao.buscarTodos().forEach(System.out::println);
 dao.atualizar(1, 3200.00);
 }
}
