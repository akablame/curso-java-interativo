O que é um Banco de Dados?

Fundação
Analogia: Planilha do Excel x Banco de Dados

Um banco de dados é como várias planilhas do Excel interligadas. Mas muito mais poderoso: suporta milhões de linhas, acesso simultâneo de vários usuários, busca ultra-rápida e relacionamentos complexos.

ColunaCabeçalho da colunanome, email, nota
Linha/RegistroUma linha de dados1 registro = 1 aluno

Criando um banco e tabela

SQL
-- Criar banco de dados
CREATE DATABASE escola;

-- Usar o banco
USE escola;

-- Criar tabela de alunos
CREATE TABLE alunos (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(100) NOT NULL,
 email VARCHAR(150) UNIQUE,
 nota DECIMAL(4,2) DEFAULT 0.0,
 turma VARCHAR(10),
 criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

SELECT — Consultando Dados

Read
Analogia: Filtro da planilha

SELECT é como usar o filtro da planilha — você escolhe quais colunasver e quais linhas(com WHERE), e pode ordenar ou limitar resultados.

SQL
-- Tudo de todos os alunos
SELECT * FROM alunos;

-- Só nome e nota
SELECT nome, nota FROM alunos;

-- Com filtro (WHERE)
SELECT nome, nota FROM alunos WHERE nota >= 7.0;

-- Com ordenação (ORDER BY)
SELECT nome, nota FROM alunos ORDER BY nota DESC;

-- Limitar resultados (os 5 melhores)
SELECT nome, nota FROM alunos ORDER BY nota DESC LIMIT 5;

-- Busca com LIKE (contém "Silva")
SELECT * FROM alunos WHERE nome LIKE '%Silva%';

-- Múltiplas condições
SELECT * FROM alunos WHERE nota >= 5 AND turma = '3A';

-- Funções de agregação
SELECT COUNT(*) AS total FROM alunos;
SELECT AVG(nota) AS media FROM alunos;
SELECT MAX(nota) AS maior, MIN(nota) AS menor FROM alunos;

-- GROUP BY (média por turma)
SELECT turma, AVG(nota) AS media, COUNT(*) AS qtd
FROM alunos
GROUP BY turma
HAVING AVG(nota) >= 6.0
ORDER BY media DESC;

UPDATE — Atualizando Dados

Update
SEMPRE use WHERE no UPDATE!

Um UPDATE sem WHERE atualiza TODOS os registros! Sempre filtre com WHERE para modificar apenas os registros desejados.

SQL
-- Atualizar nota de um aluno específico
UPDATE alunos SET nota = 10.0 WHERE id = 1;

-- Atualizar múltiplas colunas
UPDATE alunos SET nota = 9.5, turma = '3A' WHERE nome = 'Carlos Lima';

-- Atualizar com cálculo (dar +1 ponto para todos da turma 3A)
UPDATE alunos SET nota = nota + 1.0 WHERE turma = '3A' AND nota <= 9.0;

-- PERIGOSO — sem WHERE, atualiza TODOS!
-- UPDATE alunos SET nota = 10; -- NUNCA FAÇA ISSO SEM QUERER!

DELETE — Deletando Dados

Delete
SEMPRE use WHERE no DELETE!

Um DELETE sem WHERE apaga TODA A TABELA! Teste sua cláusula WHERE com SELECT antes de deletar.

SQL
-- Deletar aluno por ID (mais seguro!)
DELETE FROM alunos WHERE id = 4;

-- Deletar por condição
DELETE FROM alunos WHERE nota < 4.0;

-- Deletar com múltiplas condições
DELETE FROM alunos WHERE nota < 5 AND turma = '3B';

-- Dica: TESTE ANTES com SELECT!
SELECT * FROM alunos WHERE nota < 4.0; -- ver o que vai ser deletado
DELETE FROM alunos WHERE nota < 4.0; -- aí sim deletar

Chaves Primárias e Estrangeiras

Relacionamentos
Analogia: CPF e Matrícula

PRIMARY KEY (PK)= CPF — identificador único de cada pessoa. Nunca se repete, nunca é nulo.
FOREIGN KEY (FK)= matrícula na turma — referência a outra tabela. Liga as tabelas.

SQL — Chaves e Relacionamentos
-- Tabela TURMAS (pai)
CREATE TABLE turmas (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(10) NOT NULL, -- ex: '3A', '3B'
 ano INT NOT NULL
);

-- Tabela ALUNOS (filho) com FK para turmas
CREATE TABLE alunos (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(100) NOT NULL,
 email VARCHAR(150) UNIQUE NOT NULL,
 nota DECIMAL(4,2) DEFAULT 0.0,
 turma_id INT,
 FOREIGN KEY (turma_id) REFERENCES turmas(id)
 -- ON DELETE SET NULL -- se turma for deletada, turma_id vira NULL
 -- ON DELETE CASCADE -- se turma for deletada, alunos também são deletados
);

-- Inserir dados
INSERT INTO turmas (nome, ano) VALUES ('3A', 2024), ('3B', 2024);
INSERT INTO alunos (nome, email, nota, turma_id) VALUES
 ('Ana Silva', 'ana@email.com', 9.5, 1),
 ('Carlos Lima', 'carlos@email.com', 6.0, 2);

JOIN — Unindo Tabelas

Consulta Avançada

INNER JOIN: retorna somente os registros que têm correspondência em ambas as tabelas.

SQL
-- Alunos COM suas turmas (só quem tem turma cadastrada)
SELECT a.nome, a.nota, t.nome AS turma
FROM alunos a
INNER JOIN turmas t ON a.turma_id = t.id
ORDER BY a.nota DESC;

-- Resultado:
-- Ana Silva | 9.5 | 3A
-- Carlos Lima | 6.0 | 3B

LEFT JOIN: retorna todos da tabela da ESQUERDA + correspondências da direita (NULL se não houver).

SQL
-- TODOS os alunos, mesmo sem turma cadastrada
SELECT a.nome, a.nota, t.nome AS turma
FROM alunos a
LEFT JOIN turmas t ON a.turma_id = t.id;

-- Resultado inclui alunos sem turma (NULL):
-- Ana Silva | 9.5 | 3A
-- Carlos Lima | 6.0 | 3B
-- Lucas | 7.0 | NULL (sem turma!)

RIGHT JOIN: retorna todos da tabela da DIREITA + correspondências da esquerda.

SQL
-- TODAS as turmas, mesmo as sem alunos
SELECT a.nome, a.nota, t.nome AS turma
FROM alunos a
RIGHT JOIN turmas t ON a.turma_id = t.id;

-- Inclui turmas sem nenhum aluno:
-- Ana | 9.5 | 3A
-- Carlos | 6.0 | 3B
-- NULL | NULL | 3C (turma vazia!)

O que é modelagem?

Antes de criar as tabelas, você faz um "mapa" do banco de dados: quais entidades existem, quais são seus atributos, e como elas se relacionam. Isso evita retrabalho!

SQL — Script Completo Biblioteca
CREATE DATABASE biblioteca;
USE biblioteca;

CREATE TABLE autores (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(100) NOT NULL,
 nacionalidade VARCHAR(50)
);

CREATE TABLE livros (
 id INT PRIMARY KEY AUTO_INCREMENT,
 titulo VARCHAR(200) NOT NULL,
 isbn VARCHAR(20) UNIQUE,
 autor_id INT,
 disponivel BOOLEAN DEFAULT TRUE,
 FOREIGN KEY (autor_id) REFERENCES autores(id)
);

CREATE TABLE usuarios (
 id INT PRIMARY KEY AUTO_INCREMENT,
 nome VARCHAR(100) NOT NULL,
 email VARCHAR(150) UNIQUE NOT NULL,
 telefone VARCHAR(20)
);

CREATE TABLE emprestimos (
 id INT PRIMARY KEY AUTO_INCREMENT,
 livro_id INT NOT NULL,
 usuario_id INT NOT NULL,
 data_saida DATE NOT NULL,
 data_prevista DATE NOT NULL,
 data_retorno DATE,
 FOREIGN KEY (livro_id) REFERENCES livros(id),
 FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);

-- Inserir dados de exemplo

 ('Robert C. Martin', 'Americano'),
 ('Joshua Bloch', 'Americano');

INSERT INTO livros (titulo, isbn, autor_id) VALUES
 ('Clean Code', '978-0132350884', 1),
 ('Java Efetivo', '978-0134685991', 2);

-- Consulta: livros com nome do autor
SELECT l.titulo, a.nome AS autor, l.disponivel
FROM livros l
INNER JOIN autores a ON l.autor_id = a.id;

Java + MySQL no NetBeans

Integração
Pré-requisitos

1. MySQL instalado e rodando (porta 3306)
2. Driver JDBC adicionado ao projeto NetBeans: mysql-connector-j-8.x.x.jar
3. Banco de dados criado no MySQL Workbench

Passo 1: Adicionar o Driver JDBC no NetBeans

Botão direito no projeto → Properties → Libraries → Add JAR/Folder → Selecionar o arquivo mysql-connector-j.jar

Passo 2: Classe de Conexão

Java — Conexão MySQL
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class Conexao {
 // Dados de conexão (substitua pelos seus)
 private static final String URL = "jdbc:mysql://localhost:3306/escola";
 private static final String USUARIO = "root";
 private static final String SENHA = "suaSenha123";

 public static Connection conectar() {
 Connection conn = null;
 try {
 conn = DriverManager.getConnection(URL, USUARIO, SENHA);
 System.out.println(" Conectado ao MySQL com sucesso!");
 } catch (SQLException e) {
 System.out.println(" Erro ao conectar: " + e.getMessage());
 }
 return conn;
 }

 public static void fechar(Connection conn) {
 if (conn != null) {
 try {
 conn.close();
 System.out.println(" Conexão fechada.");
 } catch (SQLException e) {
 System.out.println("Erro ao fechar: " + e.getMessage());
 }
 }
 }
}

Passo 3: CRUD Completo em Java

Java — CRUD com JDBC
import java.sql.*;

public class AlunoDAO {
 // CREATE — Inserir aluno
 public void inserir(String nome, String email, double nota) {
 String sql = "INSERT INTO alunos (nome, email, nota) VALUES (?, ?, ?)";
 try (Connection conn = Conexao.conectar();
 PreparedStatement ps = conn.prepareStatement(sql)) {

 ps.setString(1, nome);
 ps.setString(2, email);
 ps.setDouble(3, nota);
 ps.executeUpdate();
 System.out.println(" Aluno inserido: " + nome);

 } catch (SQLException e) {
 System.out.println(" Erro: " + e.getMessage());
 }
 }

 // READ — Listar todos
 public void listarTodos() {
 String sql = "SELECT * FROM alunos ORDER BY nome";
 try (Connection conn = Conexao.conectar();
 Statement stmt = conn.createStatement();
 ResultSet rs = stmt.executeQuery(sql)) {

 System.out.println("=== ALUNOS ===");
 while (rs.next()) {
 int id = rs.getInt("id");
 String nome = rs.getString("nome");
 double nota = rs.getDouble("nota");
 System.out.printf("%d | %-20s | %.1f%n", id, nome, nota);
 }

 } catch (SQLException e) {
 System.out.println(" Erro: " + e.getMessage());
 }
 }

 // UPDATE — Atualizar nota
 public void atualizarNota(int id, double novaNota) {
 String sql = "UPDATE alunos SET nota = ? WHERE id = ?";
 try (Connection conn = Conexao.conectar();
 PreparedStatement ps = conn.prepareStatement(sql)) {

 ps.setDouble(1, novaNota);
 ps.setInt(2, id);
 int linhas = ps.executeUpdate();
 System.out.println(" " + linhas + " registro(s) atualizado(s).");

 } catch (SQLException e) {
 System.out.println(" Erro: " + e.getMessage());
 }
 }

 // DELETE — Remover aluno
 public void deletar(int id) {
 String sql = "DELETE FROM alunos WHERE id = ?";
 try (Connection conn = Conexao.conectar();
 PreparedStatement ps = conn.prepareStatement(sql)) {

 ps.setInt(1, id);
 ps.executeUpdate();
 System.out.println(" Aluno id=" + id + " removido.");

 } catch (SQLException e) {
 System.out.println(" Erro: " + e.getMessage());
 }
 }
}
PreparedStatement vs Statement

Sempre use PreparedStatementcom ?para valores de usuário. Isso previne SQL Injection— um dos ataques mais comuns em aplicações web.
