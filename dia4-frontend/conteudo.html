HTML — Estrutura das Páginas

HyperText Markup Language

CSS — Estilização Visual

Cascading Style Sheets

Seletores CSS

CSS
/* Seletor de TAG (aplica a TODOS os h1) */
h1 { color: purple; }

/* Seletor de CLASSE (aplica a todos com class="btn") */
.btn { padding: 0.5rem 1rem; }

/* Seletor de ID (aplica só ao id="header") */
#header { background: navy; }

/* Pseudo-classes */
a:hover { color: red; } /* quando passar o mouse */
input:focus { border: blue; } /* quando estiver selecionado */
button:active { opacity: 0.8; } /* quando clicar */

/* Descendente (p dentro de .card) */
.card p { font-size: 0.9rem; }

/* Combinando */
.btn.primary { background: purple; color: white; }

JavaScript — Interatividade

ECMAScript
Analogia: Eletricidade da casa

Se HTML é a estrutura e CSS é a decoração, JavaScript é a eletricidade— faz os interruptores funcionarem, acende as luzes, move as coisas.

JavaScript — DOM e Eventos
// 1. Selecionar elementos HTML
const titulo = document.getElementById('titulo');
const botoes = document.querySelectorAll('.btn');
const input = document.querySelector('input[type="text"]');

// 2. Modificar conteúdo
titulo.textContent = 'Novo Título!';
titulo.innerHTML = '<strong>Negrito</strong>';
titulo.style.color = 'purple';
titulo.classList.add('ativo');
titulo.classList.remove('inativo');

// 3. Ler valores
const valor = input.value; // texto digitado
console.log('Digitou:', valor);

// 4. Eventos
const btn = document.getElementById('meuBotao');

btn.addEventListener('click', function() {
 alert('Clicou no botão!');
});

// Evento em input
input.addEventListener('input', function() {
 console.log('Digitando:', this.value);
});

// 5. Criar elementos dinamicamente
const li = document.createElement('li');
li.textContent = 'Novo item';
document.getElementById('lista').appendChild(li);

// 6. Fetch (requisição básica)
fetch('https://api.exemplo.com/dados')
 .then(response =>response.json())
 .then(data =>console.log(data))
 .catch(error =>console.error('Erro:', error));

Exemplo Prático: Formulário com Validação

JavaScript
function validarFormulario() {
 const nome = document.getElementById('nome').value.trim();
 const email = document.getElementById('email').value.trim();
 const msgEl = document.getElementById('mensagem');

 // Limpar mensagem anterior
 msgEl.textContent = '';
 msgEl.className = '';

 // Validações
 if (nome.length < 3) {
 msgEl.textContent = ' Nome deve ter pelo menos 3 caracteres!';
 msgEl.style.color = 'red';
 return false;
 }

 if (!email.includes('@') || !email.includes('.')) {
 msgEl.textContent = ' E-mail inválido!';
 msgEl.style.color = 'red';
 return false;
 }

 // Tudo ok!
 msgEl.textContent = ' Cadastro realizado com sucesso!';
 msgEl.style.color = 'green';
 return true;
}

JFrame no NetBeans — GUI Java

JFrame é como uma janelade programa (como o Word ou Calculadora do Windows). Você coloca botões, campos de texto e labels nessa janela visualmente no NetBeans.

Formulário de Cadastro com JFrame

Java — JFrame Completo
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class TelaCadastro extends JFrame {

 // Componentes
 private JLabel lblNome, lblEmail, lblTurma;
 private JTextField txtNome, txtEmail;
 private JComboBox<String> cmbTurma;
 private JButton btnSalvar, btnLimpar;
 private JTextArea txaLista;

 public TelaCadastro() {
 // Configurar janela
 setTitle(" Cadastro de Alunos");
 setSize(500, 400);
 setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 setLocationRelativeTo(null); // centralizar na tela
 setLayout(new GridLayout(0, 2, 8, 8));
 setBackground(new Color(248, 246, 252));

 // Criar componentes
 lblNome = new JLabel("Nome:");
 txtNome = new JTextField();

 lblEmail = new JLabel("E-mail:");
 txtEmail = new JTextField();

 lblTurma = new JLabel("Turma:");
 String[] turmas = {"3A", "3B", "3C", "4A", "4B"};
 cmbTurma = new JComboBox<>(turmas);

 btnSalvar = new JButton(" Salvar");
 btnSalvar.setBackground(new Color(124, 92, 191));
 btnSalvar.setForeground(Color.WHITE);
 btnSalvar.setFont(new Font("Arial", Font.BOLD, 12));

 btnLimpar = new JButton(" Limpar");

 // Adicionar à janela
 add(lblNome); add(txtNome);
 add(lblEmail); add(txtEmail);
 add(lblTurma); add(cmbTurma);
 add(btnLimpar); add(btnSalvar);

 // ActionListener — o que acontece ao clicar
 btnSalvar.addActionListener(new ActionListener() {
 @Override
 public void actionPerformed(ActionEvent e) {
 String nome = txtNome.getText().trim();
 String email = txtEmail.getText().trim();
 String turma = (String) cmbTurma.getSelectedItem();

 if (nome.isEmpty() || email.isEmpty()) {
 JOptionPane.showMessageDialog(null,
 " Preencha todos os campos!",
 "Atenção",
 JOptionPane.WARNING_MESSAGE);
 } else {
 JOptionPane.showMessageDialog(null,
 " Aluno cadastrado!\n" + nome + " | " + email + " | " + turma,
 "Sucesso",
 JOptionPane.INFORMATION_MESSAGE);
 txtNome.setText("");
 txtEmail.setText("");
 }
 }
 });

 btnLimpar.addActionListener(e ->{
 txtNome.setText("");
 txtEmail.setText("");
 });
 }

 public static void main(String[] args) {
 // Usar Look and Feel do sistema
 try {
 UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
 } catch (Exception e) { }

 SwingUtilities.invokeLater(() ->{
 new TelaCadastro().setVisible(true);
 });
 }
}
