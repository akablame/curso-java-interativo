Calculadora Web com JavaScript
JavaScript

Crie uma calculadora web com HTML, CSS e JavaScript. Deve ter: operações básicas (+, -, *, /), display do resultado, botão de limpar e validação contra divisão por zero.

Abra o Editor ao Vivoe carregue o exemplo "Calculadora" para ver em ação!

JavaScript — Calculadora Simples
function calcular() {
 const num1 = parseFloat(document.getElementById('num1').value);
 const num2 = parseFloat(document.getElementById('num2').value);
 const op = document.getElementById('operacao').value;
 const resultEl = document.getElementById('resultado');

 if (isNaN(num1) || isNaN(num2)) {
 resultEl.textContent = ' Digite números válidos!';
 resultEl.style.color = 'red';
 return;
 }

 let resultado;
 switch (op) {
 case '+': resultado = num1 + num2; break;
 case '-': resultado = num1 - num2; break;
 case '*': resultado = num1 * num2; break;
 case '/':
 if (num2 === 0) {
 resultEl.textContent = ' Erro: divisão por zero!';
 resultEl.style.color = 'red';
 return;
 }
 resultado = num1 / num2;
 break;
 }

 resultEl.textContent = `${num1} ${op} ${num2} = ${resultado}`;
 resultEl.style.color = 'green';
}

Formulário com Validação Completa
Validação

Crie um formulário de cadastro de alunocom: nome, email, data de nascimento, curso (select) e nota (0-10). Valide todos os campos antes de "enviar" e mostre mensagem de sucesso estilizada.

JavaScript — Validação
function validar(e) {
 e.preventDefault();
 const erros = [];

 const nome = document.getElementById('nome').value.trim();
 const email = document.getElementById('email').value.trim();
 const nota = parseFloat(document.getElementById('nota').value);
 const curso = document.getElementById('curso').value;

 if (nome.length < 3) erros.push('Nome deve ter pelo menos 3 caracteres');
 if (!email.match(/^[^\s@]+@[^\s@]+\.[^\s@]+$/)) erros.push('E-mail inválido');
 if (isNaN(nota) || nota < 0 || nota >10) erros.push('Nota deve ser entre 0 e 10');
 if (!curso) erros.push('Selecione um curso');

 const erroEl = document.getElementById('erros');
 const sucessoEl = document.getElementById('sucesso');

 if (erros.length >0) {
 erroEl.innerHTML = erros.map(e =>` ${e}`).join('
');
 erroEl.style.display = 'block';
 sucessoEl.style.display = 'none';
 } else {
 erroEl.style.display = 'none';
 sucessoEl.innerHTML = ` Aluno cadastrado!
${nome}| ${email}
Curso: ${curso} | Nota: ${nota}`;
 sucessoEl.style.display = 'block';
 document.getElementById('form').reset();
 }
}

JFrame — Sistema de Login
Java Swing

Crie uma tela de logincom JFrame usando: campo de usuário, campo de senha (JPasswordField), botão entrar e botão limpar. Valide login "admin" e senha "1234", mostrando JOptionPane de sucesso ou erro.

Java — JFrame Login
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class TelaLogin extends JFrame {
 private JTextField txtUsuario;
 private JPasswordField txtSenha;
 private JButton btnEntrar, btnLimpar;
 private JLabel lblTentativas;
 private int tentativas = 0;

 public TelaLogin() {
 setTitle(" Login do Sistema");
 setSize(380, 250);
 setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 setLocationRelativeTo(null);
 setLayout(new GridLayout(5, 2, 8, 8));
 setBackground(Color.WHITE);
 ((JPanel)getContentPane()).setBorder(BorderFactory.createEmptyBorder(15,20,15,20));

 add(new JLabel(" Usuário:"));
 txtUsuario = new JTextField();
 add(txtUsuario);

 add(new JLabel(" Senha:"));
 txtSenha = new JPasswordField();
 add(txtSenha);

 lblTentativas = new JLabel("Tentativas: 0/3");
 lblTentativas.setForeground(Color.GRAY);
 add(lblTentativas);
 add(new JLabel());

 btnLimpar = new JButton(" Limpar");
 btnEntrar = new JButton("Entrar →");
 btnEntrar.setBackground(new Color(124, 92, 191));
 btnEntrar.setForeground(Color.WHITE);
 btnEntrar.setFont(new Font("Arial", Font.BOLD, 13));

 add(btnLimpar);
 add(btnEntrar);

 btnEntrar.addActionListener(e ->verificarLogin());
 txtSenha.addActionListener(e ->verificarLogin()); // Enter na senha
 btnLimpar.addActionListener(e ->{ txtUsuario.setText(""); txtSenha.setText(""); });

 setVisible(true);
 }

 private void verificarLogin() {
 if (tentativas >= 3) {
 JOptionPane.showMessageDialog(this, " Conta bloqueada!", "Bloqueado", JOptionPane.ERROR_MESSAGE);
 return;
 }

 String usuario = txtUsuario.getText().trim();
 String senha = new String(txtSenha.getPassword());

 tentativas++;
 lblTentativas.setText("Tentativas: " + tentativas + "/3");
 if (tentativas >= 3) lblTentativas.setForeground(Color.RED);

 if (usuario.equals("admin") && senha.equals("1234")) {
 JOptionPane.showMessageDialog(this,
 " Login realizado!\nBem-vindo, " + usuario + "!",
 "Sucesso", JOptionPane.INFORMATION_MESSAGE);
 dispose();
 } else {
 int restantes = 3 - tentativas;
 JOptionPane.showMessageDialog(this,
 " Usuário ou senha incorretos!\n" + restantes + " tentativa(s) restante(s).",
 "Erro", JOptionPane.ERROR_MESSAGE);
 txtSenha.setText("");
 }
 }

 public static void main(String[] args) {
 SwingUtilities.invokeLater(TelaLogin::new);
 }
}
