Cheatsheet SQL

Referência rápida de SQL e JDBC para Java

DDL — Criar Estruturas
CREATE TABLE
CREATE TABLE alunos (
 id INT AUTO_INCREMENT PRIMARY KEY,
 nome VARCHAR(100) NOT NULL,
 email VARCHAR(100) UNIQUE,
 cpf CHAR(14) UNIQUE NOT NULL,
 nota DECIMAL(4,2) DEFAULT 0.0,
 ativo BOOLEAN DEFAULT TRUE,
 criado_em DATETIME DEFAULT NOW()
);
Tipos de dados
-- Inteiros
INT / INTEGER -- -2 bi a 2 bi
BIGINT -- maior
TINYINT -- 0 a 255
SMALLINT -- -32k a 32k

-- Decimais
DECIMAL(10,2) -- 8 dígitos, 2 decimais
FLOAT / DOUBLE -- menos preciso

-- Texto
VARCHAR(n) -- variável até n chars
CHAR(n) -- fixo n chars
TEXT -- texto longo

-- Data/Hora
DATE -- 2024-03-15
TIME -- 14:30:00
DATETIME -- 2024-03-15 14:30:00
TIMESTAMP -- com timezone
Constraints (restrições)
PRIMARY KEY -- identificador único
NOT NULL -- obrigatório
UNIQUE -- sem duplicatas
DEFAULT valor -- valor padrão
FOREIGN KEY -- referência outra tabela
CHECK (nota >= 0) -- validação

-- Foreign Key completa:
FOREIGN KEY (turma_id)
 REFERENCES turmas(id)
 ON DELETE CASCADE -- ou SET NULL
 ON UPDATE CASCADE;
ALTER TABLE
-- Adicionar coluna
ALTER TABLE alunos ADD telefone VARCHAR(20);

-- Modificar tipo de coluna
ALTER TABLE alunos MODIFY nota DECIMAL(5,2);

-- Renomear coluna
ALTER TABLE alunos RENAME COLUMN nota TO media;

-- Remover coluna
ALTER TABLE alunos DROP COLUMN telefone;

-- Adicionar índice
CREATE INDEX idx_nome ON alunos(nome);
SELECT — Consultas
SELECT básico + filtros
SELECT * FROM alunos;

SELECT nome, nota FROM alunos
WHERE nota >= 7
 AND ativo = TRUE
ORDER BY nota DESC
LIMIT 10;
Operadores WHERE
WHERE nota = 8.5 -- igual
WHERE nota != 5 -- diferente
WHERE nota >= 7 -- maior ou igual
WHERE nota BETWEEN 5 AND 7 -- intervalo
WHERE nome LIKE 'Ana%' -- começa com Ana
WHERE nome LIKE '%Silva' -- termina com Silva
WHERE id IN (1, 3, 5) -- um dos valores
WHERE email IS NULL -- sem valor
WHERE email IS NOT NULL -- tem valor
GROUP BY + Aggregate
SELECT turma_id,
 COUNT(*) AS total_alunos,
 AVG(nota) AS media_turma,
 MAX(nota) AS maior_nota,
 MIN(nota) AS menor_nota,
 SUM(nota) AS soma_notas
FROM alunos
GROUP BY turma_id
HAVING AVG(nota) >= 6.5
ORDER BY media_turma DESC;
Subconsultas
-- Alunos com nota acima da média
SELECT nome, nota FROM alunos
WHERE nota >(SELECT AVG(nota) FROM alunos);

-- EXISTS
SELECT * FROM pedidos p
WHERE EXISTS (
 SELECT 1 FROM itens i WHERE i.pedido_id = p.id
);
JOINs
INNER JOIN
-- Só registros que têm correspondência nas duas
SELECT a.nome, t.nome AS turma
FROM alunos a
INNER JOIN turmas t ON a.turma_id = t.id;
LEFT JOIN
-- Todos da esquerda, NULL onde não há correspondência
SELECT a.nome, e.data_emprestimo
FROM usuarios a
LEFT JOIN emprestimos e ON a.id = e.usuario_id;
-- Usuários sem empréstimos terão NULL em e.*
Múltiplos JOINs
SELECT a.nome, t.nome AS turma,
 p.nome AS professor, n.nota
FROM alunos a
JOIN turmas t ON a.turma_id = t.id
JOIN professores p ON t.professor_id = p.id
LEFT JOIN notas n ON n.aluno_id = a.id
ORDER BY t.nome, a.nome;
INSERT / UPDATE / DELETE
INSERT
-- Um registro
INSERT INTO alunos (nome, nota)
VALUES ('Maria', 8.5);

-- Múltiplos registros
INSERT INTO alunos (nome, nota) VALUES
('João', 7.0),
('Ana', 9.5),
('Pedro', 6.0);

-- Retornar id gerado (MySQL)
INSERT INTO ... VALUES ...;
SELECT LAST_INSERT_ID();
UPDATE 
-- SEMPRE use WHERE para não atualizar tudo!
UPDATE alunos
SET nota = 9.0, ativo = TRUE
WHERE id = 5;

-- Atualizar baseado em outra tabela
UPDATE alunos a
JOIN turmas t ON a.turma_id = t.id
SET a.nota = a.nota + 0.5
WHERE t.nome = 'Turma A';
DELETE 
-- SEMPRE use WHERE!
DELETE FROM alunos WHERE id = 5;

-- Deletar com JOIN
DELETE a FROM alunos a
JOIN turmas t ON a.turma_id = t.id
WHERE t.nome = 'Turma Cancelada';

-- Deletar todos (sem WHERE) — perigo!
TRUNCATE TABLE log_acesso; -- rápido, sem rollback
DELETE FROM log_acesso; -- lento, com rollback

CASE (condicional)
SELECT nome,
 CASE
 WHEN nota >= 7.0 THEN 'Aprovado'
 WHEN nota >= 5.0 THEN 'Recuperação'
 ELSE 'Reprovado'
 END AS situacao
FROM alunos;

-- CASE simples
SELECT nome,
 CASE ativo
 WHEN TRUE THEN 'Sim'
 ELSE 'Não'
 END AS esta_ativo
FROM alunos;
