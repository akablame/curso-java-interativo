Sistema de Biblioteca

Um sistema completo de gerenciamento de biblioteca com empréstimos, devoluções e controle de usuários.

Java + MySQLDificuldade: ~3 dias de desenvolvimento
Requisitos Funcionais
  • RF01: Cadastrar/editar/excluir livros
  • RF02: Cadastrar usuários (nome, CPF, e-mail)
  • RF03: Realizar empréstimo de livro
  • RF04: Registrar devolução
  • RF05: Listar livros disponíveis
  • RF06: Consultar histórico de empréstimos
  • RF07: Calcular multa por atraso (R$1/dia)
Tecnologias

  • MySQL 8.0
  • JFrame (Swing)
  • Arquitetura MVC
  • DAO Pattern
  • PreparedStatement (segurança)

Script SQL

SQL — Biblioteca
CREATE DATABASE IF NOT EXISTS biblioteca DEFAULT CHARACTER SET utf8mb4;
USE biblioteca;

CREATE TABLE livros (
 id INT AUTO_INCREMENT PRIMARY KEY,
 titulo VARCHAR(200) NOT NULL,
 autor VARCHAR(100) NOT NULL,
 isbn VARCHAR(20) UNIQUE,
 ano_publicacao INT,
 quantidade INT DEFAULT 1,
 disponivel BOOLEAN DEFAULT TRUE
);

CREATE TABLE usuarios (
 id INT AUTO_INCREMENT PRIMARY KEY,
 nome VARCHAR(100) NOT NULL,
 cpf VARCHAR(14) UNIQUE NOT NULL,
 email VARCHAR(100),
 data_cadastro DATE DEFAULT (CURRENT_DATE)
);

CREATE TABLE emprestimos (
 id INT AUTO_INCREMENT PRIMARY KEY,
 livro_id INT NOT NULL,
 usuario_id INT NOT NULL,
 data_emprestimo DATE NOT NULL,
 data_devolucao DATE,
 devolvido BOOLEAN DEFAULT FALSE,
 multa DECIMAL(10,2) DEFAULT 0.00,
 FOREIGN KEY (livro_id) REFERENCES livros(id),
 FOREIGN KEY (usuario_id) REFERENCES usuarios(id)
);

public class Emprestimo {
 private int id;
 private Livro livro;
 private Usuario usuario;
 private LocalDate dataEmprestimo, dataDevolucao;
 private boolean devolvido;
 private double multa;

 public double calcularMulta() {
 if (devolvido || dataDevolucao == null) return 0;
 long diasAtraso = ChronoUnit.DAYS.between(dataDevolucao, LocalDate.now());
 return diasAtraso >0 ? diasAtraso * 1.0 : 0;
 }
}

// === DAO ===
public class LivroDAO {
 private Connection conn;
 public LivroDAO(Connection conn) { this.conn = conn; }

 public void inserir(Livro l) throws SQLException {
 String sql = "INSERT INTO livros(titulo,autor,isbn,ano_publicacao,quantidade) VALUES(?,?,?,?,?)";
 PreparedStatement ps = conn.prepareStatement(sql);
 ps.setString(1, l.getTitulo());
 ps.setString(2, l.getAutor());
 ps.setString(3, l.getIsbn());
 ps.setInt(4, l.getAnoPublicacao());
 ps.setInt(5, l.getQuantidade());
 ps.executeUpdate();
 }

 public List<Livro> listarDisponiveis() throws SQLException {
 List<Livro> lista = new ArrayList<>();
 String sql = "SELECT * FROM livros WHERE disponivel = TRUE";
 ResultSet rs = conn.createStatement().executeQuery(sql);
 while (rs.next()) {
 Livro l = new Livro();
 l.setId(rs.getInt("id"));
 l.setTitulo(rs.getString("titulo"));
 l.setAutor(rs.getString("autor"));
 lista.add(l);
 }
 return lista;
 }
}

// === CONTROLLER ===
public class EmprestimoController {
 private LivroDAO livroDAO;
 private EmprestimoDAO empDAO;

 public void realizarEmprestimo(int livroId, int usuarioId, LocalDate devPrev) throws Exception {
 Livro livro = livroDAO.buscarPorId(livroId);
 if (!livro.isDisponivel()) throw new Exception("Livro indisponível!");

 Emprestimo emp = new Emprestimo();
 emp.setLivro(livro);
 emp.setDataEmprestimo(LocalDate.now());
 emp.setDataDevolucao(devPrev);
 empDAO.inserir(emp);

 livroDAO.setDisponivel(livroId, false);
 }
}

SQL — Sistema Escolar (queries principais)
-- Boletim do aluno com médias
SELECT a.nome, d.nome AS disciplina,
 n.nota1, n.nota2,
 (n.nota1 + n.nota2) / 2 AS media,
 CASE WHEN (n.nota1 + n.nota2) / 2 >= 7 THEN 'Aprovado'
 WHEN (n.nota1 + n.nota2) / 2 >= 5 THEN 'Recuperação'
 ELSE 'Reprovado' END AS situacao
FROM notas n
JOIN alunos a ON n.aluno_id = a.id
JOIN disciplinas d ON n.disciplina_id = d.id
WHERE a.id = ?
ORDER BY d.nome;

-- Ranking da turma
SELECT a.nome, AVG((n.nota1 + n.nota2)/2) AS media_geral
FROM alunos a
JOIN notas n ON n.aluno_id = a.id
WHERE a.turma_id = ?
GROUP BY a.id, a.nome
ORDER BY media_geral DESC;

-- Trigger para calcular situação automaticamente
DELIMITER //
CREATE TRIGGER calcular_situacao
BEFORE INSERT ON notas
FOR EACH ROW BEGIN
 SET NEW.media = (NEW.nota1 + NEW.nota2) / 2;
 SET NEW.situacao = CASE
 WHEN NEW.media >= 7.0 THEN 'Aprovado'
 WHEN NEW.media >= 5.0 THEN 'Recuperação'
 ELSE 'Reprovado'
 END;
END //
DELIMITER ;

Java — Controller de Vendas
public class VendaController {
 private VeiculoDAO veicDAO;
 private ClienteDAO cliDAO;
 private VendaDAO vendaDAO;

 public Venda realizarVenda(int veicId, int cliId, double descPct, String forma) throws Exception {
 Veiculo v = veicDAO.buscarPorId(veicId);
 if (!v.isDisponivel()) throw new Exception("Veículo já vendido!");

 Cliente c = cliDAO.buscarPorId(cliId);

 double valorFinal = v.getPreco() * (1 - descPct / 100.0);

 // Validação: desconto máximo 15%
 if (descPct >15) throw new Exception("Desconto máximo é 15%!");

 Venda venda = new Venda(v, c, LocalDate.now(), valorFinal, descPct, forma);
 vendaDAO.inserir(venda);

 veicDAO.marcarComoVendido(veicId);
 return venda;
 }

 public double calcularFaturamentoMes(int mes, int ano) throws SQLException {
 return vendaDAO.somarVendasPorMes(mes, ano);
 }
}

SQL — Queries de Estoque
-- Produtos abaixo do estoque mínimo (ALERTA!)
SELECT codigo, nome, qtd_estoque, qtd_minima,
 (qtd_minima - qtd_estoque) AS precisa_repor
FROM produtos
WHERE qtd_estoque <= qtd_minima
ORDER BY precisa_repor DESC;

-- Relatório de movimentação do mês
SELECT p.nome, m.tipo, SUM(m.quantidade) AS total,
 COUNT(*) AS num_movimentos
FROM movimentacoes m
JOIN produtos p ON m.produto_id = p.id
WHERE MONTH(m.data_mov) = MONTH(CURRENT_DATE)
 AND YEAR(m.data_mov) = YEAR(CURRENT_DATE)
GROUP BY p.id, p.nome, m.tipo
ORDER BY p.nome;

-- Procedure para registrar saída (com validação)
DELIMITER //
CREATE PROCEDURE registrar_saida(IN p_id INT, IN p_qtd INT, IN p_usuario VARCHAR(50))
BEGIN
 DECLARE estoque_atual INT;
 SELECT qtd_estoque INTO estoque_atual FROM produtos WHERE id = p_id;
 IF estoque_atual < p_qtd THEN
 SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Estoque insuficiente!';
 END IF;
 UPDATE produtos SET qtd_estoque = qtd_estoque - p_qtd WHERE id = p_id;
 INSERT INTO movimentacoes(produto_id, tipo, quantidade, data_mov, usuario)
 VALUES(p_id, 'saida', p_qtd, NOW(), p_usuario);
END //
DELIMITER ;

Java — Ordem de Serviço (Controller principal)
public class OrdemServicoController {

 public OrdemServico abrirOS(int veiculoId, String problemaRelatado) throws SQLException {
 OrdemServico os = new OrdemServico();
 os.setVeiculo(veiculoDAO.buscarPorId(veiculoId));
 os.setDataEntrada(LocalDate.now());
 os.setStatus("Aguardando Diagnóstico");
 os.setProblemaRelatado(problemaRelatado);
 osDAO.inserir(os);
 return os;
 }

 public void adicionarItem(int osId, int pecaId, int qtd, String tipo) throws Exception {
 Peca peca = pecaDAO.buscarPorId(pecaId);
 if (tipo.equals("peça") && peca.getEstoque() < qtd) {
 throw new Exception("Estoque insuficiente: " + peca.getNome());
 }
 ItemOS item = new ItemOS(osId, pecaId, qtd, peca.getPreco(), tipo);
 itemDAO.inserir(item);
 if (tipo.equals("peça")) pecaDAO.reduzirEstoque(pecaId, qtd);
 }

 public double calcularTotal(int osId) throws SQLException {
 return itemDAO.somarItensPorOS(osId);
 }

 public void fecharOS(int osId) throws SQLException {
 double total = calcularTotal(osId);
 osDAO.atualizar(osId, "Pronto", LocalDate.now(), total);
 }
}
